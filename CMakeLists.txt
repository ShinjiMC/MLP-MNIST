cmake_minimum_required(VERSION 3.10)
project(perceptron)

set(CMAKE_CXX_STANDARD 17)

# Agregar flags de optimizaci√≥n y OpenMP
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3 -fopenmp")

include_directories(include)

# Ejecutable de entrenamiento
add_executable(mlp_train

     src/layer.cpp
     src/mlp.cpp

     src/dataset.cpp
     src/config.cpp
     src/regularizer.cpp
)

add_executable(mlp_test
     main_test.cpp
     src/layer.cpp
     src/mlp.cpp
     src/neuron.cpp
     src/dataset.cpp
     src/config.cpp
     src/regularizer.cpp
)
